<script>

 function refreshShowPwd() {
  if ($('#showPwdCB').is(':checked')) {
   $('#passphrase').prop('type', 'text')
  } else {
   $('#passphrase').prop('type', 'password')
   //$('#passphrase').type = 'text'
    //document.getElementById('passphrase').type = 'text';
  }
 }

 var dft_koi = '10000';
 var koi = dft_koi;

 $(document).ready(function () {
  showMain();
 });

 function config_ok() {
  koi = $('#nzr_koi').val();
  showMain();
 }

 function config_reset() {
  $('#nzr_koi').val(dft_koi);
 }

 function clearAllFields() {
  $('#mainText').val('');
  $('#passphrase').val('');
  showMain();
 }

 function scramble() {
  $('#nzr_errors').val('');
  $('#nzr_errors').hide();
  var sci = {}
  sci["text"] = $("#mainText").val();
  sci["koi"] = koi;
  sci["passphrase"] = $("#passphrase").val();
  $.ajax({
   type: "POST",
   contentType: "application/json",
   url: "/beam/scramble",
   data: JSON.stringify(sci),
   cache: false,
   timeout: 10000,
   success: function(data) {
    $('#mainText').val(data)
   },
   error: function (j, e) {
    showErrors(j, e);
   }
  });
 }

 function unscramble() {
  $('#nzr_errors').val('');
  $('#nzr_errors').hide();
  var sci = {}
  sci["text"] = $("#mainText").val();
  sci["koi"] = koi;
  sci["passphrase"] = $("#passphrase").val();
  $.ajax({
   type: "POST",
   contentType: "application/json",
   url: "/beam/unscramble",
   data: JSON.stringify(sci),
   cache: false,
   timeout: 10000,
   success: function(data) {
    $('#mainText').val(data)
   },
   error: function (j, e) {
    showErrors(j, e);
   }
  });
 } 

 function showErrors(j, e) {
  msg = formatErrors(j, e);
  $('#nzr_error').html(msg);
  $('#nzr_errors').show();
  $('#nzr_help').hide();
  $('#nzr_config').hide();
  $('#nzr_main_000').show();
  $('#nzr_main_001').show();
 }

 
 function showMain() {
  $('#nzr_errors').hide();
  $('#nzr_help').hide();
  $('#nzr_config').hide();
  $('#nzr_main_000').show();
  $('#nzr_main_001').show();
  refreshShowPwd();
 }
 
 function showHelp() {
  $('#nzr_errors').hide();
  $('#nzr_main_000').hide();
  $('#nzr_main_001').hide();
  $('#nzr_config').hide();
  $('#nzr_help').show();
 }
 
 function showConfig() {
  $('#nzr_errors').hide();
  $('#nzr_koi').val(koi);
  $('#nzr_help').hide();
  $('#nzr_main_000').hide();
  $('#nzr_main_001').hide();
  $('#nzr_config').show();
 }



</script>
